Uji Coba Perangkat

Gunakan kode tes ini untuk memastikan kabel terpasang dengan benar.

#include 
#include 
#include "Adafruit_BME680.h"

Adafruit_BME680 bme;

void setup() {
  Serial.begin(115200);
  Serial.println("Mulai Tes Sensor...");
  if (!bme.begin()) {
    Serial.println("Sensor Tidak Terdeteksi! Periksa Kabel.");
    while (1);
  }
}

void loop() {
  if (! bme.performReading()) return;
  Serial.print("Suhu: "); Serial.println(bme.temperature);
  delay(2000);
}
